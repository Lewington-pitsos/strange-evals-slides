% Score-based color: maps [25.8,96.2] via red-yellow-green gradient
\newcommand{\SC}[1]{%
  \pgfmathsetmacro{\clamped}{min(96.2, max(25.8, #1))}%
  \pgfmathsetmacro{\pctval}{(\clamped - 25.8) / 0.704}%
  \pgfmathsetmacro{\finalR}{min(1, 2*(100-\pctval)/100)*0.35 + 0.65}%
  \pgfmathsetmacro{\finalG}{min(1, 2*\pctval/100)*0.35 + 0.65}%
  \edef\applycolor{\noexpand\cellcolor[rgb]{\finalR,\finalG,0.65}}%
  \applycolor #1%
}
